# ============================================================
# Workspace Manager - Python Dependencies
# ============================================================
# Install all dependencies with:
#   pip install -r requirements.txt
#
# Compatible with Python 3.12, 3.13, and 3.14.
# ============================================================

# Speech recognition engine -- converts audio to text.
# Uses Google's free Web Speech API by default (requires internet).
SpeechRecognition>=3.11.0

# Audio capture -- records from the microphone as NumPy arrays.
# Ships with pre-built PortAudio binaries, so unlike PyAudio it
# installs cleanly on Python 3.12 - 3.14 without a C compiler.
sounddevice>=0.5.0

# WAV file I/O -- writes recorded audio to a temp .wav file
# that SpeechRecognition can read.
scipy>=1.13.0

# Text-to-speech engine -- optional, for future voice feedback
# (e.g. confirming which workspace was launched).
pyttsx3>=2.98

# ============================================================
# Voice Overlay Dependencies (added for --overlay mode)
# ============================================================

# Offline speech recognition engine (Vosk/Kaldi) -- PRIMARY.
# Works without internet after a one-time ~40 MB model download.
vosk>=0.3.45

# Fast fuzzy string matching for typo-tolerant command recognition.
# "oppen dashbord" still matches "open dashboard".
rapidfuzz>=3.6.0

# Global hotkey support -- lets Ctrl+Shift+M activate the mic
# from any application.
keyboard>=0.13.5

# Image processing -- used for enhanced overlay icon rendering.
Pillow>=10.0.0

# Optional: high-accuracy offline speech (CTranslate2 Whisper).
# Uncomment to enable:  pip install faster-whisper
